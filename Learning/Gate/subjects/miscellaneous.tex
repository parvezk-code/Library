            In the below function, which of the following is the correct expression for E? (GATE-2006)

    \begin{lstlisting}
        int func(int m, int n)
        {
            if(E) return 1;
            else
            {
                return(func(m -1, n) + func(m - 1, n - 1));
            }
        }
    \end{lstlisting}

  \begin{choices}
    \CorrectChoice  (n = = 0) OR  (m = = 1)
    \choice         (n = = 0) AND (m = = 1)
    \choice         (n = = 0) OR  (m = = n)
    \choice         (n = = 0) AND (m = = n)
  \end{choices}
\end{questyle}

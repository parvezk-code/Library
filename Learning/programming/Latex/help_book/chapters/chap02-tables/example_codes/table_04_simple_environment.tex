% usepackage : xparse(NewDocumentEnvironment), [table]{xcolor}, float(h!), caption
\NewDocumentEnvironment{simpleCenteringTempChapterTable}{m} % #1 = caption
{%
  \begingroup                         % <--- start local scope
  \begin{table}[!h]
    \centering
    \setlength{\tabcolsep}{10pt}
    \renewcommand{\arraystretch}{1.4}
}{%
    \caption{#1}
  \end{table}
  \endgroup                           % <--- end local scope (cleans variables)
}

\begin{simpleCenteringTempChapterTable}{Table with environment}
    \def\cellOneA{\textbf{Fruit}}
    \def\cellOneB{\textcolor{red}{Apple}}
    \def\cellTwoA{\textbf{Animal}}
    \def\cellTwoB{\textcolor{orange}{Cat}}

    \begin{tabular}{|c|c|} \hline
        \cellOneA    & \cellOneB     \\ \hline
        \cellTwoA    & \cellTwoB		\\ \hline
    \end{tabular}
\end{simpleCenteringTempChapterTable}

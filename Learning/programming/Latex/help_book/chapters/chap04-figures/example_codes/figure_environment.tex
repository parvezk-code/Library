\NewDocumentEnvironment{simpleImage}{ m m o o } 
{
  \begin{figure}[h!]
    \centering
    \includegraphics[width=#2]{#1} 
    \IfValueTF{#3}{\caption{#3}}{}
    \IfValueTF{#4}{\label{#4}}{}
}
{
  \end{figure}
}

	{0.6\linewidth}
